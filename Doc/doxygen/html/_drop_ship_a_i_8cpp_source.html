57  obj_w = self->Location() + Point(0, 1e3, 0);

59

61  else {

63  Point dst = dst_rgn->GetOrbitalRegion()->Location() -
64  self_rgn->GetOrbitalRegion()->Location() +

66 
67  obj_w = dst.OtherHand();
68  }
69 
70  // distance from self to navpt:

72
