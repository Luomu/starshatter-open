185 void EditBox::EnsureCaretVisible()
186 {
187  if (!single_line) {
188  h_offset = 0;
189  return;
190  }
191 
192  if (sel_start < 0) {
193  sel_start = 0;
194  h_offset = 0;
195  }
196 
197  else if (sel_start > h_offset) {
198  int x_caret;
199  bool moved;
200 
201  do {
202  x_caret = 0;
203  moved = false;
204 
205  if (pass_char) {
206  Text pass = Text(pass_char, sel_start-h_offset);
207  x_caret += font->StringWidth(pass.data(), pass.length());
208  }
209  else {
210  Text sub = text.substring(h_offset, sel_start-h_offset);
211  x_caret += font->StringWidth(sub.data(), sub.length());
212  }
213 
214  if (x_caret >= Width()-4) {
215  if (h_offset < text.length()) {
216  h_offset++;
217  moved = true;
218  }
219  }
220  }
221  while (moved);
222  }
223 
224  else {
225  h_offset = sel_start;
226  }
227 }

313 void EditBox::Insert(char c)
314 {
315  if (single_line && c == '\n')
316  return;
317 
318  if (sel_start < text.length()) {
319  if (sel_start == 0) {
320  text = Text(c) + text;
321  sel_start = 1;
322  }
323  else {
324  Text t1 = text.substring(0, sel_start);
325  Text t2 = text.substring(sel_start, text.length()-sel_start);
326  text = t1 + Text(c) + t2;
327  sel_start++;
328  }
329  }
330  else {
331  text += c;
332  sel_start = text.length();
333  }
334 
335  EnsureCaretVisible();
336 }
