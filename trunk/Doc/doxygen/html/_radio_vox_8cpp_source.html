215 bool
216 RadioVox::Update()
217 {
218  if (message.length()) {
219  RadioView::Message(message);
220  message = "";
221  }
222 
223  bool active = false;
224 
225  while (!active && index < sounds.size()) {
226  Sound* s = sounds[index];
227 
228  if (s->IsReady()) {
229  if (channel & 1)
230  s->SetPan(channel * -3000);
231  else
232  s->SetPan(channel * 3000);
233 
234  s->Play();
235  active = true;
236  }
237 
238  else if (s->IsPlaying()) {
239  s->Update();
240  active = true;
241  }
242 
243  else {
244  index++;
245  }
246  }
247 
248  return active;
