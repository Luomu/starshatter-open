138 void
139 MsnSelectDlg::Show()
140 {
141  FormWindow::Show();
142  campaign = Campaign::GetCampaign();
143 
144  if (cmb_campaigns) {
145  int n = 0;
146  cmb_campaigns->ClearItems();
147  ListIter<Campaign> iter = Campaign::GetAllCampaigns();
148  while (++iter) {
149  Campaign* c = iter.value();
150 
151  if (c->GetCampaignId() >= Campaign::SINGLE_MISSIONS) {
152  cmb_campaigns->AddItem(c->Name());
153 
154  if (campaign->GetCampaignId() < Campaign::SINGLE_MISSIONS) {
155  campaign = Campaign::SelectCampaign(c->Name());
156  cmb_campaigns->SetSelection(n);
157  }
158 
159  else if (campaign->GetCampaignId() == c->GetCampaignId()) {
160  cmb_campaigns->SetSelection(n);
161  }
162 
163  n++;
164  }
165  }
166  }
167 
168  else if (lst_campaigns) {
169  int n = 0;
170  lst_campaigns->ClearItems();
171  ListIter<Campaign> iter = Campaign::GetAllCampaigns();
172  while (++iter) {
173  Campaign* c = iter.value();
174 
175  if (c->GetCampaignId() >= Campaign::SINGLE_MISSIONS) {
176  lst_campaigns->AddItem(c->Name());
177 
178  if (campaign->GetCampaignId() < Campaign::SINGLE_MISSIONS) {
179  campaign = Campaign::SelectCampaign(c->Name());
180  lst_campaigns->SetSelected(n);
181  }
182 
183  else if (campaign->GetCampaignId() == c->GetCampaignId()) {
184  lst_campaigns->SetSelected(n);
185  }
186 
187  n++;
188  }
189  }
190  }
191 
192  if (campaign) {
193  int id = campaign->GetCampaignId();
194  editable = (id >= Campaign::MULTIPLAYER_MISSIONS &&
195  id <= Campaign::CUSTOM_MISSIONS);
196 
197  if (btn_new) btn_new->SetEnabled(editable);
198  if (btn_edit) btn_edit->SetEnabled(false);
199  if (btn_del) btn_del->SetEnabled(false);
200  }
201 
202  if (description)
203  description->SetText(Game::GetText("MsnSelectDlg.choose"));
204 
205  if (lst_missions) {
206  lst_missions->ClearItems();
207 
208  if (campaign) {
209  ListIter<MissionInfo> iter = campaign->GetMissionList();
210  while (++iter) {
211  MissionInfo* info = iter.value();
212  Mission* m = info->mission;
213 
214  lst_missions->AddItem(info->name);
215 
216  if (m && m == edit_mission) {
217  lst_missions->SetSelected(lst_missions->NumItems()-1);
218  }
219  }
220 
221  if (selected_mission >= 0 && lst_missions->GetSelCount() == 0) {
222  lst_missions->SetSelected(selected_mission);
223  }
224  }
225 
226  OnMissionSelect(0);
227  edit_mission = 0;
228  }
229 }

363 void
364 MsnSelectDlg::OnNew(AWEvent* event)
365 {
366  const char* cname = 0;
367 
368  if (cmb_campaigns)
369  cname = cmb_campaigns->GetSelectedItem();
370  else if (lst_campaigns)
371  cname = lst_campaigns->GetSelectedItem();
372 
373  Campaign* c = Campaign::SelectCampaign(cname);
374  if (!c) return;
375 
376  MissionInfo* info = c->CreateNewMission();
377  if (!info || !info->mission)
378  return;
379 
380  mission_id = info->id;
381 
382  MsnEditDlg* editor = manager->GetMsnEditDlg();
383 
384  if (editor) {
385  edit_mission = info->mission;
386 
387  editor->SetMissionInfo(info);
388  editor->SetMission(info->mission);
389  manager->ShowMsnEditDlg();
390  }
391 
392  MsnEditNavDlg* navdlg = (MsnEditNavDlg*) manager->GetNavDlg();
393 
394  if (navdlg) {
395  navdlg->SetMission(info->mission);
396  navdlg->SetMissionInfo(info);
397  }
